                        <div style={{ marginTop:22, fontSize:11, color:"rgba(255,255,255,0.15)" }}>нажми чтобы перевернуть</div>
                      </>
                    ) : (
                      <>
                        <div style={{ fontSize:10, color:"rgba(52,211,153,0.5)", letterSpacing:2, textTransform:"uppercase", marginBottom:14 }}>Перевод</div>
                        <div style={{ fontSize:26, fontWeight:700, color:"#34d399", textAlign:"center", lineHeight:1.4 }}>{word.ru}</div>
                        <div style={{ marginTop:18, fontSize:36, fontWeight:900, color:"rgba(255,255,255,0.1)" }}>{word.kr}</div>
                      </>
                    )}
                  </div>
                  <div style={{ display:"flex", gap:10, marginTop:20, width:"100%" }}>
                    <button onClick={prevCard} style={{ flex:1, padding:"15px", borderRadius:12, border:"1px solid rgba(255,255,255,0.08)", background:"rgba(255,255,255,0.04)", color:"#fff", fontSize:18, cursor:"pointer" }}>←</button>
                    <button onClick={() => setFlipped(f=>!f)} style={{ flex:2, padding:"15px", borderRadius:12, border:"none", background:"rgba(255,255,255,0.04)", color:"rgba(255,255,255,0.4)", fontSize:12, cursor:"pointer", fontFamily:"inherit" }}>перевернуть</button>
                    <button onClick={nextCard} style={{ flex:1, padding:"15px", borderRadius:12, border:"1px solid rgba(255,255,255,0.08)", background:"rgba(255,255,255,0.04)", color:"#fff", fontSize:18, cursor:"pointer" }}>→</button>
                  </div>
                </div>
              </div>
            );
          }

          // QUIZ
          if (screen === "quiz") {
            const word = quizWords[quizIndex];
            const qpct = Math.round((quizIndex/quizWords.length)*100);
            return (
              <div style={{ minHeight:"100vh", background:"#0a0a0f", fontFamily:"'Noto Sans KR',sans-serif", display:"flex", flexDirection:"column" }}>
                <link href="https://fonts.googleapis.com/css2?family=Noto+Sans+KR:wght@300;400;700;900&display=swap" rel="stylesheet"/>
                <div style={{ display:"flex", alignItems:"center", padding:"20px 16px 12px", gap:10 }}>
                  <button onClick={() => setScreen("home")} style={{ background:"rgba(255,255,255,0.07)", border:"none", borderRadius:10, color:"#fff", padding:"8px 14px", cursor:"pointer", fontSize:13, fontFamily:"inherit" }}>← Назад</button>
                  <div style={{ flex:1, height:3, borderRadius:2, background:"rgba(255,255,255,0.08)", overflow:"hidden" }}>
                    <div style={{ height:"100%", width:`${qpct}%`, background:"#7c3aed", transition:"width 0.3s" }}/>
                  </div>
                  <div style={{ color:"#34d399", fontSize:13, fontWeight:700, whiteSpace:"nowrap" }}>{score}✓</div>
                </div>
                <div style={{ flex:1, display:"flex", flexDirection:"column", padding:"0 16px 20px" }}>
                  <div style={{ borderRadius:18, padding:"28px 20px", textAlign:"center", marginBottom:16, background:"linear-gradient(135deg,#12122a,#1a1040)", border:"1px solid rgba(124,58,237,0.2)" }}>
                    <div style={{ fontSize:10, color:"rgba(255,255,255,0.25)", letterSpacing:2, marginBottom:8, textTransform:"uppercase" }}>{quizIndex+1}/{quizWords.length} · {CAT_LABELS[word.category]}</div>
                    <div style={{ fontSize:13, color:"rgba(255,255,255,0.4)", marginBottom:6 }}>Напиши по-корейски:</div>
                    <div style={{ fontSize:32, fontWeight:900, color:"#fff", lineHeight:1.2 }}>{word.ru}</div>
                    {showRomanji && <div style={{ fontSize:13, color:"rgba(255,255,255,0.2)", marginTop:8, fontStyle:"italic" }}>[{word.romanji}]</div>}
                  </div>
                  <input ref={inputRef} value={input} onChange={e=>setInput(e.target.value)}
                    onKeyDown={e=>{ if(e.key==="Enter") answered ? nextQuiz() : checkAnswer(); }}
                    disabled={!!answered} placeholder="한국어로 써요..."
                    style={{ width:"100%", padding:"16px", borderRadius:14, boxSizing:"border-box", background:answered==="correct"?"rgba(52,211,153,0.12)":answered==="wrong"?"rgba(239,68,68,0.12)":"rgba(255,255,255,0.07)", color:"#fff", fontSize:28, fontWeight:700, fontFamily:"inherit", textAlign:"center", outline:"none", border:answered==="correct"?"2px solid rgba(52,211,153,0.4)":answered==="wrong"?"2px solid rgba(239,68,68,0.4)":"2px solid transparent", transition:"all 0.2s" }}
                  />
                  {answered && (
                    <div style={{ marginTop:12, padding:"14px 16px", borderRadius:12, textAlign:"center", background:answered==="correct"?"rgba(52,211,153,0.1)":"rgba(239,68,68,0.1)", border:`1px solid ${answered==="correct"?"rgba(52,211,153,0.25)":"rgba(239,68,68,0.25)"}` }}>
                      {answered==="correct"
                        ? <div style={{ color:"#34d399", fontWeight:700, fontSize:17 }}>✓ Правильно!</div>
                        : <><div style={{ color:"#f87171", fontWeight:700, fontSize:15 }}>✗ Неправильно</div><div style={{ color:"rgba(255,255,255,0.35)", fontSize:12, marginTop:3 }}>Правильный ответ:</div><div style={{ color:"#fff", fontWeight:900, fontSize:30, marginTop:2 }}>{word.kr}</div></>}
                    </div>
                  )}
                  <div style={{ flex:1 }}/>
                  <button onClick={answered ? nextQuiz : checkAnswer} style={{ width:"100%", padding:"17px", borderRadius:14, border:"none", background:answered?"linear-gradient(135deg,#34d399,#059669)":"linear-gradient(135deg,#7c3aed,#5b21b6)", color:"#fff", fontSize:16, fontWeight:700, cursor:"pointer", fontFamily:"inherit", boxShadow:answered?"0 4px 20px rgba(52,211,153,0.25)":"0 4px 20px rgba(124,58,237,0.35)" }}>
                    {answered ? (quizIndex+1>=quizWords.length ? "Результат →" : "Дальше →") : "Проверить ✓"}
                  </button>
                </div>
              </div>
            );
          }

          // RESULT
          if (screen === "result") {
            const total = quizWords.length;
            const pctScore = Math.round((score/total)*100);
            return (
              <div style={{ minHeight:"100vh", background:"#0a0a0f", fontFamily:"'Noto Sans KR',sans-serif", display:"flex", flexDirection:"column", alignItems:"center", padding:"36px 16px 40px" }}>
                <link href="https://fonts.googleapis.com/css2?family=Noto+Sans+KR:wght@300;400;700;900&display=swap" rel="stylesheet"/>
                <div style={{ fontSize:56 }}>{pctScore>=80?"🔥":pctScore>=60?"💪":"📚"}</div>
                <div style={{ fontSize:24, fontWeight:900, color:"#fff", marginTop:8 }}>{score} / {total}</div>
                <div style={{ fontSize:44, fontWeight:900, color:pctScore>=70?"#34d399":"#f87171", margin:"4px 0" }}>{pctScore}%</div>
                <div style={{ fontSize:13, color:"rgba(255,255,255,0.35)", marginBottom:24 }}>{pctScore>=70?"Отлично! 화이팅! 💪":"Повтори ошибки и попробуй снова!"}</div>
                {mistakes.length>0 && (
                  <div style={{ width:"100%", marginBottom:20 }}>
                    <div style={{ fontSize:11, color:"rgba(255,255,255,0.3)", marginBottom:10, letterSpacing:1, textTransform:"uppercase" }}>Ошибки ({mistakes.length})</div>
                    <div style={{ display:"flex", flexDirection:"column", gap:7 }}>
                      {mistakes.map((m,i) => (
                        <div key={i} style={{ background:"rgba(239,68,68,0.07)", border:"1px solid rgba(239,68,68,0.18)", borderRadius:12, padding:"10px 14px", display:"flex", justifyContent:"space-between", alignItems:"center" }}>
                          <div><div style={{ color:"rgba(255,255,255,0.35)", fontSize:11 }}>{m.ru}</div><div style={{ color:"#fff", fontWeight:700, fontSize:20 }}>{m.kr}</div></div>
                          <div style={{ textAlign:"right" }}><div style={{ color:"rgba(255,255,255,0.25)", fontSize:10 }}>Ты написал:</div><div style={{ color:"#f87171", fontWeight:700 }}>{m.typed||"—"}</div></div>
                        </div>
                      ))}
                    </div>
                  </div>
                )}
                <div style={{ display:"flex", flexDirection:"column", gap:10, width:"100%" }}>
                  <button onClick={startQuiz} style={{ width:"100%", padding:"16px", borderRadius:13, border:"none", background:"linear-gradient(135deg,#7c3aed,#5b21b6)", color:"#fff", fontSize:15, fontWeight:700, cursor:"pointer", fontFamily:"inherit" }}>🔄 Попробовать снова</button>
                  <button onClick={() => setScreen("home")} style={{ width:"100%", padding:"16px", borderRadius:13, border:"1px solid rgba(255,255,255,0.1)", background:"rgba(255,255,255,0.05)", color:"#fff", fontSize:15, fontWeight:700, cursor:"pointer", fontFamily:"inherit" }}>← На главную</button>
                </div>
              </div>
            );
          }
        }

        // Рендерим приложение в root-директорию
        const root = WebpackReactDOM = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
